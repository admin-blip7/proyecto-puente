Add a `loadRecentNotifications` async function inside the component:

```typescript
const loadRecentNotifications = async () => {
  if (!selectedBranch?.id) return;
  const { createClient } = await import('@supabase/supabase-js');
  // Use the existing supabase client from the project — check existing imports first.
  // If the page already imports supabase, use that. Otherwise import from '@/lib/supabaseClient'.
  // The query is:
  const { data } = await supabase
    .from('whatsapp_notification_log')
    .select('id, status, sent_at, phone_to, error_msg')
    .eq('branch_id', selectedBranch.id)
    .order('sent_at', { ascending: false })
    .limit(5);
  setRecentNotifications(data ?? []);
};
```

<Card>
